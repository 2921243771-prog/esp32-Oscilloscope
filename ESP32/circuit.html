软件换算

float readBipolarVoltage(int adcPin) {
    float vout = analogRead(adcPin) / 4095.0 * 3.3;
    return (1.65 - vout) * 6.0;  // 反相还原为 ±10V
}

// 使用:
float ch1 = readBipolarVoltage(4);  // GPIO4 → CH1
float ch2 = readBipolarVoltage(5);  // GPIO5 → CH2
